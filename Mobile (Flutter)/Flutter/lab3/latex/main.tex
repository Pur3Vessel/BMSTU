Исходный код программы представлен в листинге~\ref{lst:code1}

\begin{figure}[H]
\begin{lstlisting}[language={},caption={Реализация мобильного приложения},label={lst:code1}]
import 'package:flutter/material.dart';
import 'package:flutter_cube/flutter_cube.dart';

void main() {
  runApp(MyApp());
}

class MyApp extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      home: MyHomePage(),
    );
  }
}

class MyHomePage extends StatefulWidget {
  @override
  _MyHomePageState createState() => _MyHomePageState();
}

class _MyHomePageState extends State<MyHomePage> {
  double skullOffset = -0.55;
  late Scene _scene;
  late Object _skullObject;

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: const Text('Pain'),
      ),
      body: Column(
        mainAxisAlignment: MainAxisAlignment.center,
        children: [
          Expanded(
            child: Cube(
              onSceneCreated: (Scene scene) {
                _scene = scene;
                _scene.camera.position.z = 5;
                _scene.camera.position.y = 10;
                _scene.camera.position.x = 0;
                _skullObject = Object(scale: Vector3(11, 11, 11), fileName: 'assets/piece/12140_Skull_v3_L2.obj', position: Vector3(-0.01, -0.97, -0.55));
                scene.world.add(Object(scale: Vector3(10, 10, 10), fileName: 'assets/skull/12140_Skull_v3_L2.obj'));
                scene.world.add(_skullObject);
              },
            ),
          ),
          Slider(
            value: skullOffset,
            onChanged: (value) {
              setState(() {
                skullOffset = value;
                _skullObject.position.x = -0.01;
                _skullObject.position.y = -0.97;
                _skullObject.position.z = skullOffset;
                _skullObject.updateTransform();
              });
            },
            min: -0.55,
            max: 0.25,
          ),
        ],
      ),
    );
  }
}
\end{lstlisting}
\end{figure}
