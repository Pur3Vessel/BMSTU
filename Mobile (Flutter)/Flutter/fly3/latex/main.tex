\section{Задание}\label{Sect::task}
1. Реализовать мобильное приложение управления двумя переменными на сайте ioControl.ru

2. Для выполнения задачи использовать пакет 'http/dart'.

Исходный код программы представлен в листинге~\ref{lst:code1}

\begin{figure}[H]
\begin{lstlisting}[language={},caption={Реализация мобильного приложения},label={lst:code1}]
import 'dart:html';

import 'package:flutter/material.dart';
import 'package:http/http.dart' as http;

void main() {
  runApp(const MyApp());
}

class MyApp extends StatelessWidget {
  const MyApp({super.key});

  // This widget is the root of your application.
  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      title: 'Flutter Demo',
      theme: ThemeData(
        colorScheme: ColorScheme.fromSeed(seedColor: Colors.deepPurple),
        useMaterial3: true,
      ),
      home: const MyHomePage(title: 'Flutter Demo Home Page'),
    );
  }
}

class MyHomePage extends StatefulWidget {
  const MyHomePage({super.key, required this.title});
  final String title;

  @override
  State<MyHomePage> createState() => _MyHomePageState();
}

class _MyHomePageState extends State<MyHomePage> {
  int _counter = 0;

  void _incrementCounter() {
    setState(() {
      _counter++;
    });
  }

  void _getLampRequestON() { 
    setState(() {
      http.get(Uri.parse('http://iocontrol.ru/api/sendData/Fly_mobile/Lamp/1')).then((response) {
        print("Response status: ${response.statusCode}");
        print("Response body: ${response.body}");
      }).catchError((error){
        print("Error: $error");
      });
    });
  }
  void _getLampRequestOFF() {
    // <----- Step 2
    setState(() {
      http.get(Uri.parse('http://iocontrol.ru/api/sendData/Fly_mobile/Lamp/0'))
          .then((response) {
        print("Response status: ${response.statusCode}");
        print("Response body: ${response.body}");
      }).catchError((error) {
        print("Error: $error");
      });
    });
  }
    void _getDoorRequestOFF() {
      // <----- Step 2
      setState(() {
        http.get(
            Uri.parse('http://iocontrol.ru/api/sendData/Fly_mobile/Door/0'))
            .then((response) {
          print("Response status: ${response.statusCode}");
          print("Response body: ${response.body}");
        }).catchError((error) {
          print("Error: $error");
        });
      });
    }
  void _getDoorRequestON() {
    // <----- Step 2
    setState(() {
      http.get(
          Uri.parse('http://iocontrol.ru/api/sendData/Fly_mobile/Door/1'))
          .then((response) {
        print("Response status: ${response.statusCode}");
        print("Response body: ${response.body}");
      }).catchError((error) {
        print("Error: $error");
      });
    });
  }
  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: Text(widget.title),
      ),
      body: Center(
          mainAxisAlignment: MainAxisAlignment.center,
          children: <Widget>[
            const Text(
              'You have pushed the button this many times:',
            ),
            Text(
              '$_counter',
              style: Theme.of(context).textTheme.headlineMedium,
            ),
            TextButton(
              style: ButtonStyle(
                foregroundColor: MaterialStateProperty.all<Color>(Colors.blue),
              ),
              onPressed: _getLampRequestON,
              child: Text('ON_LAMP'),
            ),
            TextButton( 
              style: ButtonStyle(
                foregroundColor: MaterialStateProperty.all<Color>(Colors.blue),
              ),
              onPressed: _getLampRequestOFF,
              child: Text('OFF_LAMP'),
            ),
            TextButton(
              style: ButtonStyle(
                foregroundColor: MaterialStateProperty.all<Color>(Colors.blue),
              ),
              onPressed: _getDoorRequestON,
              child: Text('ON_DOOR'),
            ),
            TextButton( 
              style: ButtonStyle(
                foregroundColor: MaterialStateProperty.all<Color>(Colors.blue),
              ),
              onPressed: _getDoorRequestOFF,
              child: Text('OFF_DOOR'),
            )
          ],
        ),
      ),
      floatingActionButton: FloatingActionButton(
        onPressed: _incrementCounter,
        tooltip: 'Increment',
        child: const Icon(Icons.add),
      ), // This trailing comma makes auto-formatting nicer for build methods.
    );

  }
}
